            const data = await response.json();

                return data.candidates[0].content.parts[0].text;

            } else {

                // This case handles situations where the response is valid but empty, e.g., due to safety settings.

                console.warn("Gemini API returned a response with no content.", data);

                return null;

            }

        } catch (error) {

            console.error("Error calling Gemini API:", error);

            return null; // Return null to indicate failure

        }

    }

    function calculateROI() {

        const h = parseFloat(document.getElementById('heures').value);

        const c = parseFloat(document.getElementById('collaborateurs').value);

        const t = parseFloat(document.getElementById('cout').value);

        

        document.getElementById('heures-val').textContent = h;

        document.getElementById('collaborateurs-val').textContent = c;

        document.getElementById('cout-val').textContent = t;


        const economie = h * c * t * 12;

        document.getElementById('roi-result').textContent = `${economie.toLocaleString(currentLang)} €`;

        updateRoiChart(economie);

    }
